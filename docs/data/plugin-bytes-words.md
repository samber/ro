import (
    "github.com/samber/ro"
    robytes "github.com/samber/ro/plugins/bytes"
)

obs := ro.Pipe(
    ro.Just([]byte("hello world from go")),
    robytes.Words[[]byte](),
)

sub := obs.Subscribe(ro.PrintObserver[[][]byte]())
